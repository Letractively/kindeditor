 13 (function (K, undefined) {
 14 
 15 var _SINGLE_TAG_MAP = K._SINGLE_TAG_MAP,
 16 	_each = K.each,
 17 	_trim = K.trim,
 18 	_toHex = K.toHex,
 19 	_inString = K.inString;
 20 
 21 function _getCssList(css) {
 22 	var list = {},
 23 		reg = /\s*([\w\-]+)\s*:([^;]*)(;|$)/g,
 24 		match;
 25 	while (match = reg.exec(css)) {
 26 		var key = _trim(match[1].toLowerCase()),
 27 			val = _trim(_toHex(match[2]));
 28 		list[key] = val;
 29 	}
 30 	return list;
 31 }
 32 
 33 function _getAttrList(tag) {
 34 	var list = {},
 35 		reg = /\s+(?:([\w-:]+)|(?:([\w-:]+)=([^\s"'<>]+))|(?:([\w-:]+)="([^"]*)")|(?:([\w-:]+)='([^']*)'))(?=(?:\s|\/|>)+)/g,
 36 		match;
 37 	while (match = reg.exec(tag)) {
 38 		var key = match[1] || match[2] || match[4] || match[6],
 39 			val = (match[2] ? match[3] : (match[4] ? match[5] : match[7])) || '';
 40 		list[key] = val;
 41 	}
 42 	return list;
 43 }
 44 
 45 function _formatCss(css) {
 46 	var str = '';
 47 	_each(_getCssList(css), function(key, val) {
 48 		str += key + ':' + val + ';';
 49 	});
 50 	return str;
 51 }
 52 
 53 //Modified by zhanyi1022
 54 function _formatHtml(html) {
 55 	var re = /((?:[\r\n])*)<(\/)?([\w-:]+)(\s*(?:[\w-:]+)(?:=(?:"[^"]*"|'[^']*'|[^\s"'<>]*))?)*\s*(\/)?>((?:[\r\n])*)/g;
 56 	html = html.replace(re, function($0, $1, $2, $3, $4, $5, $6) {
 57 		var startNewline = $1 || '',
 58 			startSlash = $2 || '',
 59 			tagName = $3.toLowerCase(),
 60 			attr = $4 || '',
 61 			endSlash = $5 ? ' ' + $5 : '',
 62 			endNewline = $6 || '';
 63 		if (endSlash === '' && _SINGLE_TAG_MAP[tagName]) endSlash = ' /';
 64 		if (endNewline) endNewline = ' ';
 65 		if (tagName !== 'script' && tagName !== 'style') {
 66 			startNewline = '';
 67 		}
 68 		if (attr !== '') {
 69 			attr = attr.replace(/\s*([\w-:]+)=([^\s"'<>]+|"[^"]*"|'[^']*')/g, function($0, $1, $2) {
 70 				var key = $1.toLowerCase(),
 71 					val = $2 || '';
 72 				if (val === '') {
 73 					val = '""';
 74 				} else {
 75 					if (key === 'style') {
 76 						val = val.substr(1, val.length - 2);
 77 						val = _formatCss(val);
 78 						if (val === '') return '';
 79 						val = '"' + val + '"';
 80 					}
 81 					if (!/^['"]/.test(val)) val = '"' + val + '"';
 82 				}
 83 				return ' ' + key + '=' + val + ' ';
 84 			});
 85 			attr = _trim(attr);
 86 			attr = attr.replace(/\s+/g, ' ');
 87 			if (attr) attr = ' ' + attr;
 88 			return startNewline + '<' + startSlash + tagName + attr + endSlash + '>' + endNewline;
 89 		} else {
 90 			return startNewline + '<' + startSlash + tagName + endSlash + '>' + endNewline;
 91 		}
 92 	});
 93 	return _trim(html);
 94 }
 95 K.formatHtml = _formatHtml;
 96 K._formatCss = _formatCss;
 97 K._getCssList = _getCssList;
 98 K._getAttrList = _getAttrList;
 99 
100 })(KindEditor);
