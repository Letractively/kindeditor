 18 var _IE = K.IE,
 19 	_node = K.node,
 20 	_inArray = K.inArray,

 26 /**
 27 	Reference:
 28 	DOM Level 2: http://www.w3.org/TR/DOM-Level-2-Traversal-Range/ranges.html

 30 function _range(mixed) {
 31 	if (!mixed.nodeName) {
 32 		return mixed.get ? mixed : _toRange(mixed);
 33 	}
 34 	var doc = mixed;
 35 	return {
 36 		startContainer : doc,
 37 		startOffset : 0,
 38 		endContainer : doc,
 39 		endOffset : 0,
 40 		collapsed : true,
 41 		commonAncestorContainer : doc,
 42 		setStart : function(node, offset) {
 43 			this.startContainer = node;
 44 			this.startOffset = offset;
 45 			if (this.endContainer === doc) {
 46 				this.endContainer = node;
 47 				this.endOffset = offset;
 48 			}
 49 			_compareAndUpdate.call(this, doc);
 50 			_updateCollapsed.call(this);
 51 			_updateCommonAncestor.call(this, doc);
 52 			return this;
 53 		},
 54 		setEnd : function(node, offset) {
 55 			this.endContainer = node;
 56 			this.endOffset = offset;
 57 			if (this.startContainer === doc) {
 58 				this.startContainer = node;
 59 				this.startOffset = offset;
 60 			}
 61 			_compareAndUpdate.call(this, doc);
 62 			_updateCollapsed.call(this);
 63 			_updateCommonAncestor.call(this, doc);
 64 			return this;
 65 		},
 66 		setStartBefore : function(node) {
 67 			return this.setStart(node.parentNode || doc, _node(node).index);
 68 		},
 69 		setStartAfter : function(node) {
 70 			return this.setStart(node.parentNode || doc, _node(node).index + 1);
 71 		},
 72 		setEndBefore : function(node) {
 73 			return this.setEnd(node.parentNode || doc, _node(node).index);
 74 		},
 75 		setEndAfter : function(node) {
 76 			return this.setEnd(node.parentNode || doc, _node(node).index + 1);
 77 		},
 78 		selectNode : function(node) {
 79 			this.setStartBefore(node);
 80 			this.setEndAfter(node);
 81 			return this;
 82 		},
 83 		selectNodeContents : function(node) {
 84 			var knode = _node(node);
 85 			if (knode.type == 3 || knode.isSingle()) {
 86 				this.selectNode(node);
 87 			} else {
 88 				if (knode.children.length > 0) {
 89 					this.setStartBefore(knode.first.get());
 90 					this.setEndAfter(knode.last.get());
 91 				} else {
 92 					this.setStart(node, 0);
 93 					this.setEnd(node, 0);
 94 				}
 95 			}
 96 			return this;
 97 		},
 98 		collapse : function(toStart) {
 99 			if (toStart) this.setEnd(this.startContainer, this.startOffset);
100 			else this.setStart(this.endContainer, this.endOffset);
101 			return this;
102 		},
103 		compareBoundaryPoints : function(how, range) {
104 			var rangeA = this.get(),
105 				rangeB = range.get();
106 			if (_IE) {
107 				var arr = {};
108 				arr[_START_TO_START] = 'StartToStart';
109 				arr[_START_TO_END] = 'EndToStart';
110 				arr[_END_TO_END] = 'EndToEnd';
111 				arr[_END_TO_START] = 'StartToEnd';
112 				var cmp = rangeA.compareEndPoints(arr[how], rangeB);
113 				if (cmp !== 0) return cmp;
114 				var nodeA, nodeB, posA, posB;
115 				if (how === _START_TO_START || how === _END_TO_START) {
116 					nodeA = this.startContainer;
117 					posA = this.startOffset;
118 				}
119 				if (how === _START_TO_END || how === _END_TO_END) {
120 					nodeA = this.endContainer;
121 					posA = nodeA.nodeType == 1 ? this.endOffset - 1 : this.endOffset;
122 				}
123 				if (how === _START_TO_START || how === _START_TO_END) {
124 					nodeB = range.startContainer;
125 					posB = range.startOffset;
126 				}
127 				if (how === _END_TO_END || how === _END_TO_START) {
128 					nodeB = range.endContainer;
129 					posB = nodeB.nodeType == 1 ? range.endOffset - 1 : range.endOffset;
130 				}
131 				if (nodeA === nodeB) return 0;
132 				var childA = nodeA,
133 					childB = nodeB;
134 				if (nodeA.nodeType === 1) {
135 					childA = nodeA.childNodes[posA];
136 					if (childA === nodeB) {
137 						if (how === _START_TO_START || how === _END_TO_START) return -1;
138 						if (how === _END_TO_END || how === _START_TO_END) return 1;
139 					}
140 				}
141 				if (nodeB.nodeType === 1) {
142 					childB = nodeB.childNodes[posB];
143 					if (childB === nodeA) {
144 						if (how === _START_TO_START || how === _END_TO_START) return 1;
145 						if (how === _END_TO_END || how === _START_TO_END) return -1;
146 					}
147 				}
148 				if (childA && childB === childA.nextSibling) return -1;
149 				if (childB && childA === childB.nextSibling) return 1;
150 				var bool = _node(childB).isAncestor(childA);
151 				if (how === _START_TO_START || how === _END_TO_START) return bool ? -1 : 1;
152 				if (how === _END_TO_END || how === _START_TO_END) return bool ? 1 : -1;
153 			} else {
154 				return rangeA.compareBoundaryPoints(how, rangeB);
155 			}
156 		},
157 		cloneRange : function() {
158 			var range = _range(doc);
159 			range.setStart(this.startContainer, this.startOffset);
160 			range.setEnd(this.endContainer, this.endOffset);
161 			return range;
162 		},
163 		toString : function() {
164 			var rng = this.get(),
165 				str = _IE ? rng.text : rng.toString();
166 			return str.replace(/\r\n|\n|\r/g, '');
167 		},
168 		cloneContents : function() {
169 			return _copyAndDelete.call(this, doc, true, false);
170 		},
171 		deleteContents : function() {
172 			return _copyAndDelete.call(this, doc, false, true);
173 		},
174 		extractContents : function() {
175 			return _copyAndDelete.call(this, doc, true, true);
176 		},
177 		insertNode : function(node) {
178 			var startContainer = this.startContainer,
179 				startOffset = this.startOffset,
180 				endContainer = this.endContainer,
181 				endOffset = this.endOffset,
182 				afterNode,
183 				parentNode,
184 				endNode,
185 				endTextNode,
186 				endTextPos,
187 				eq = startContainer == endContainer,
188 				isFrag = node.nodeName.toLowerCase() === '#document-fragment';
189 			if (endContainer.nodeType == 1 && endOffset > 0) {
190 				endNode = endContainer.childNodes[endOffset - 1];
191 				if (endNode.nodeType == 3) {
192 					eq = startContainer == endNode;
193 					if (eq) endTextPos = endNode.nodeValue.length;
194 				}
195 			}
196 			if (startContainer.nodeType == 1) {
197 				if (startContainer.childNodes.length > 0) {
198 					afterNode = startContainer.childNodes[startOffset];
199 				} else {
200 					parentNode = startContainer;
201 				}
202 			} else {
203 				if (startOffset == 0) {
204 					afterNode = startContainer;
205 				} else if (startOffset < startContainer.length) {
206 					afterNode = startContainer.splitText(startOffset);
207 					if (eq) {
208 						endTextNode = afterNode;
209 						endTextPos = endTextPos ? endTextPos - startOffset : this.endOffset - startOffset;
210 						this.setEnd(endTextNode, endTextPos);
211 					}
212 				} else {
213 					if (startContainer.nextSibling) {
214 						afterNode = startContainer.nextSibling;
215 					} else {
216 						parentNode = startContainer.parentNode;
217 					}
218 				}
219 			}
220 			if (afterNode) afterNode.parentNode.insertBefore(node, afterNode);
221 			if (parentNode) parentNode.appendChild(node);
222 			if (isFrag) {
223 				if (node.firstChild) this.setStartBefore(node.firstChild);
224 				if (this.collapsed) {
225 					if (afterNode) endNode = afterNode.previousSibling;
226 					if (parentNode) endNode = parentNode.lastChild;
227 				}
228 			} else {
229 				this.setStartBefore(node);
230 				if (this.collapsed) endNode = node;
231 			}
232 			if (endNode) this.setEndAfter(endNode);
233 			return this;
234 		},
235 		surroundContents : function(node) {
236 			node.appendChild(this.extractContents());
237 			return this.insertNode(node);
238 		},
239 		get : function() {
240 			var startContainer = this.startContainer,
241 				startOffset = this.startOffset,
242 				endContainer = this.endContainer,
243 				endOffset = this.endOffset,
244 				range;
245 			if (doc.createRange) {
246 				range = doc.createRange();
247 				range.selectNodeContents(doc.body);
248 			} else {
249 				range = doc.body.createTextRange();
250 			}
251 			if (_IE) {
252 				range.setEndPoint('StartToStart', _getEndRange(startContainer, startOffset));
253 				range.setEndPoint('EndToStart', _getEndRange(endContainer, endOffset));
254 			} else {
255 				range.setStart(startContainer, startOffset);
256 				range.setEnd(endContainer, endOffset);
257 			}
258 			return range;
259 		},
260 		html : function() {
261 			//TODO
262 			return _node(this.cloneContents()).outer().toLowerCase();
263 		}
264 	};

267 function _updateCollapsed() {
268 	this.collapsed = (this.startContainer === this.endContainer && this.startOffset === this.endOffset);
269 }

271 function _updateCommonAncestor(doc) {
272 	function scan(node, fn) {
273 		if (node === doc) return;
274 		while (node) {
275 			if (fn(node)) return;
276 			node = node.parentNode;
277 		}
278 	}
279 	var nodes = [];
280 	scan(this.startContainer, function(node) {
281 		nodes.push(node);
282 	});
283 	var ancestor = doc;
284 	scan(this.endContainer, function(node) {
285 		if (_inArray(node, nodes) >= 0) {
286 			ancestor = node;
287 			return true;
288 		}
289 	});
290 	this.commonAncestorContainer = ancestor;
291 }

293 function _compareAndUpdate(doc) {
294 	var rangeA = _range(doc),
295 		rangeB = _range(doc);
296 	rangeA.startContainer = rangeA.endContainer = this.startContainer;
297 	rangeA.startOffset = rangeA.endOffset = this.startOffset;
298 	rangeB.startContainer = rangeB.endContainer = this.endContainer;
299 	rangeB.startOffset = rangeB.endOffset = this.endOffset;
300 	if (rangeA.compareBoundaryPoints(_START_TO_START, rangeB) == 1) {
301 		this.startContainer = this.endContainer;
302 		this.startOffset = this.endOffset;
303 	}
304 }

306 /*
307 	cloneContents: copyAndDelete(true, false)
308 	extractContents: copyAndDelete(true, true)
309 	deleteContents: copyAndDelete(false, true)
310 */
311 function _copyAndDelete(doc, isCopy, isDelete) {
312 	var self = this,
313 		startContainer = self.startContainer,
314 		startOffset = self.startOffset,
315 		endContainer = self.endContainer,
316 		endOffset = self.endOffset,
317 		nodeList = [],
318 		selfRange = self;
319 	if (isDelete) {
320 		selfRange = self.cloneRange();
321 		self.collapse(true);
322 		if (startContainer.nodeType == 3 && startOffset == 0) {
323 			self.setStart(startContainer.parentNode, 0);
324 			self.setEnd(startContainer.parentNode, 0);
325 		}
326 	}
327 	function splitTextNode(node, startOffset, endOffset) {
328 		var length = node.nodeValue.length,
329 			centerNode;
330 		if (isCopy) {
331 			var cloneNode = node.cloneNode(true),
332 			centerNode = cloneNode.splitText(startOffset);
333 			centerNode.splitText(endOffset - startOffset);
334 		}
335 		if (isDelete) {
336 			var center = node;
337 			if (startOffset > 0) center = node.splitText(startOffset);
338 			if (endOffset < length) center.splitText(endOffset - startOffset);
339 			nodeList.push(center);
340 		}
341 		return centerNode;
342 	}
343 	function getTextNode(node) {
344 		if (node == startContainer && node == endContainer) {
345 			return splitTextNode(node, startOffset, endOffset);
346 		} else if (node == startContainer) {
347 			return splitTextNode(node, startOffset, node.nodeValue.length);
348 		} else if (node == endContainer) {
349 			return splitTextNode(node, 0, endOffset);
350 		} else {
351 			return splitTextNode(node, 0, node.nodeValue.length);
352 		}
353 	}
354 	function extractNodes(parent, frag) {
355 		var node = parent.firstChild;
356 		while (node) {
357 			var range = _range(doc);
358 			range.selectNode(node);
359 			if (range.compareBoundaryPoints(_END_TO_START, selfRange) >= 0) return false;
360 			var nextNode = node.nextSibling;
361 			if (range.compareBoundaryPoints(_START_TO_END, selfRange) > 0) {
362 				var type = node.nodeType;
363 				if (type == 1) {
364 					if (range.compareBoundaryPoints(_START_TO_START, selfRange) >= 0) {
365 						if (isCopy) {
366 							frag.appendChild(node.cloneNode(true));
367 						}
368 						if (isDelete) {
369 							nodeList.push(node);
370 						}
371 					} else {
372 						var childFlag;
373 						if (isCopy) {
374 							childFlag = node.cloneNode(false);
375 							frag.appendChild(childFlag);
376 						}
377 						if (!extractNodes(node, childFlag)) return false;
378 					}
379 				} else if (type == 3) {
380 					var textNode = getTextNode(node);
381 					if (textNode) frag.appendChild(textNode);
382 				}
383 			}
384 			node = nextNode;
385 		}
386 		return true;
387 	}
388 	var frag = doc.createDocumentFragment(),
389 		ancestor = selfRange.commonAncestorContainer;
390 	if (ancestor.nodeType == 3) {
391 		var textNode = getTextNode(ancestor);
392 		if (textNode) frag.appendChild(textNode);
393 	} else {
394 		extractNodes(ancestor, frag);
395 	}
396 	for (var i = 0, len = nodeList.length; i < len; i++) {
397 		var node = nodeList[i];
398 		node.parentNode.removeChild(node);
399 	}
400 	return isCopy ? frag : undefined;
401 }

403 function _getStartEnd(rng, isStart) {
404 	var doc = rng.parentElement().ownerDocument;
405 	var range = _range(doc);
406 	var pointRange = rng.duplicate();
407 	pointRange.collapse(isStart);
408 	var parent = pointRange.parentElement();
409 	var children = parent.childNodes;
410 	if (children.length == 0) {
411 		range.selectNode(parent);
412 		return {node: range.startContainer, offset: range.startOffset};
413 	}
414 	var startNode = doc, startPos = 0, isEnd = false;
415 	var testRange = rng.duplicate();
416 	testRange.moveToElementText(parent);
417 	for (var i = 0, len = children.length; i < len; i++) {
418 		var node = children[i];
419 		var cmp = testRange.compareEndPoints('StartToStart', pointRange);
420 		if (cmp > 0) isEnd = true;
421 		if (cmp == 0) {
422 			var range = _range(doc);
423 			if (node.nodeType == 1) range.selectNode(node);
424 			else range.setStartBefore(node);
425 			return {node: range.startContainer, offset: range.startOffset};
426 		}
427 		if (node.nodeType == 1) {
428 			var nodeRange = rng.duplicate();
429 			nodeRange.moveToElementText(node);
430 			testRange.setEndPoint('StartToEnd', nodeRange);
431 			if (isEnd) startPos += nodeRange.text.length;
432 			else startPos = 0;
433 		} else if (node.nodeType == 3) {
434 			testRange.moveStart('character', node.nodeValue.length);
435 			startPos += node.nodeValue.length;
436 		}
437 		if (!isEnd) startNode = node;
438 	}
439 	if (!isEnd && startNode.nodeType == 1) {
440 		range.setStartAfter(parent.lastChild);
441 		return {node: range.startContainer, offset: range.startOffset};
442 	}
443 	testRange = rng.duplicate();
444 	testRange.moveToElementText(parent);
445 	testRange.setEndPoint('StartToEnd', pointRange);
446 	startPos -= testRange.text.length;
447 	return {node: startNode, offset: startPos};
448 }

450 function _toRange(rng) {
451 	if (_IE) {
452 		var doc = rng.parentElement().ownerDocument;
453 		if (rng.item) {
454 			var range = _range(doc);
455 			range.selectNode(rng.item(0));
456 			return range;
457 		}
458 		var start = _getStartEnd(rng, true),
459 			end = _getStartEnd(rng, false),
460 			range = _range(doc);
461 		range.setStart(start.node, start.offset);
462 		range.setEnd(end.node, end.offset);
463 		return range;
464 	} else {
465 		var startContainer = rng.startContainer,
466 			doc = startContainer.ownerDocument || startContainer,
467 			range = _range(doc);
468 		range.setStart(startContainer, rng.startOffset);
469 		range.setEnd(rng.endContainer, rng.endOffset);
470 		return range;
471 	}
472 }

474 function _getBeforeLength(node) {
475 	var doc = node.ownerDocument,
476 		len = 0,
477 		sibling = node.previousSibling;
478 	while (sibling) {
479 		if (sibling.nodeType == 1) {
480 			if (!_node(sibling).isSingle()) {
481 				var range = doc.body.createTextRange();
482 				range.moveToElementText(sibling);
483 				len += range.text.length;
484 			} else {
485 				len += 1;
486 			}
487 		} else if (sibling.nodeType == 3) {
488 			len += sibling.nodeValue.length;
489 		}
490 		sibling = sibling.previousSibling;
491 	}
492 	return len;
493 }

495 function _getEndRange(node, offset) {
496 	var doc = node.ownerDocument || node,
497 		range = doc.body.createTextRange();
498 	if (doc == node) {
499 		range.collapse(true);
500 		return range;
501 	}
502 	if (node.nodeType == 1) {
503 		var children = node.childNodes,
504 			isStart,
505 			child,
506 			isTemp = false;
507 		if (offset == 0) {
508 			child = children[0];
509 			isStart = true;
510 		} else {
511 			child = children[offset - 1];
512 			isStart = false;
513 		}
514 		if (!child) {
515 			var temp = doc.createTextNode(' ');
516 			node.appendChild(temp);
517 			child = temp;
518 			isTemp = true;
519 		}
520 		if (child.nodeName.toLowerCase() === 'head') {
521 			if (offset === 1) isStart = true;
522 			if (offset === 2) isStart = false;
523 			range.collapse(isStart);
524 			return range;
525 		}
526 		if (child.nodeType == 1) {
527 			range.moveToElementText(child);
528 			range.collapse(isStart);
529 		} else {
530 			range.moveToElementText(node);
531 			if (isTemp) node.removeChild(temp);
532 			var len = _getBeforeLength(child);
533 			len = isStart ? len : len + child.nodeValue.length;
534 			range.moveStart('character', len);
535 		}
536 	} else if (node.nodeType == 3) {
537 		range.moveToElementText(node.parentNode);
538 		range.moveStart('character', offset + _getBeforeLength(node));
539 	}
540 	return range;
541 }
