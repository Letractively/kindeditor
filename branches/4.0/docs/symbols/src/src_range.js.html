275 		/**
276 			@name KindEditor.range#compareBoundaryPoints
277 			@function
278 			@public
279 			@param {Int} how 位置信息，可设置K.START_TO_START、K.START_TO_END、K.END_TO_END、K.END_TO_START。
280 			@param {KRange} range 目标range
281 			@returns {Int} 当this range在目标range的左侧时返回-1，在目标range的右侧时返回1，相同时返回0。
282 			@description
283 			<p>根据how参数比较2个range的边界。</p>
284 			<p>how参数的方向规则：</p>
285 			<p>K.START_TO_START：比较目标range的开始位置和this range的开始位置。</p>
286 			<p>K.START_TO_END：比较目标range的开始位置和this range的结束位置。</p>
287 			<p>K.END_TO_END：比较目标range的结束位置和this range的结束位置。</p>
288 			<p>K.END_TO_START：比较目标range的结束位置和this range的开始位置。</p>
289 		*/
290 		compareBoundaryPoints : function(how, range) {
291 			var rangeA = this.get(),
292 				rangeB = range.get();
293 			if (_IE) {
294 				var arr = {};
295 				arr[_START_TO_START] = 'StartToStart';
296 				arr[_START_TO_END] = 'EndToStart';
297 				arr[_END_TO_END] = 'EndToEnd';
298 				arr[_END_TO_START] = 'StartToEnd';
299 				var cmp = rangeA.compareEndPoints(arr[how], rangeB);
300 				if (cmp !== 0) return cmp;
301 				var nodeA, nodeB, posA, posB;
302 				if (how === _START_TO_START || how === _END_TO_START) {
303 					nodeA = this.startContainer;
304 					posA = this.startOffset;
305 				}
306 				if (how === _START_TO_END || how === _END_TO_END) {
307 					nodeA = this.endContainer;
308 					posA = nodeA.nodeType == 1 ? this.endOffset - 1 : this.endOffset;
309 				}
310 				if (how === _START_TO_START || how === _START_TO_END) {
311 					nodeB = range.startContainer;
312 					posB = range.startOffset;
313 				}
314 				if (how === _END_TO_END || how === _END_TO_START) {
315 					nodeB = range.endContainer;
316 					posB = nodeB.nodeType == 1 ? range.endOffset - 1 : range.endOffset;
317 				}
318 				if (nodeA === nodeB) return 0;
319 				var childA = nodeA,
320 					childB = nodeB;
321 				if (nodeA.nodeType === 1) {
322 					childA = nodeA.childNodes[posA];
323 					if (childA === nodeB) {
324 						if (how === _START_TO_START || how === _END_TO_START) return -1;
325 						if (how === _END_TO_END || how === _START_TO_END) return 1;
326 					}
327 				}
328 				if (nodeB.nodeType === 1) {
329 					childB = nodeB.childNodes[posB];
330 					if (childB === nodeA) {
331 						if (how === _START_TO_START || how === _END_TO_START) return 1;
332 						if (how === _END_TO_END || how === _START_TO_END) return -1;
333 					}
334 				}
335 				if (childA && childB === childA.nextSibling) return -1;
336 				if (childB && childA === childB.nextSibling) return 1;
337 				var bool = _node(childB).isAncestor(childA);
338 				if (how === _START_TO_START || how === _END_TO_START) return bool ? -1 : 1;
339 				if (how === _END_TO_END || how === _START_TO_END) return bool ? 1 : -1;
340 			} else {
341 				return rangeA.compareBoundaryPoints(how, rangeB);
342 			}
343 		},

405 		/**
406 			@name KindEditor.range#insertNode
407 			@function
408 			@public
409 			@param {Node} node
410 			@returns {KRange}
411 			@description
412 			将指定Node插入到KRange的开始位置。
413 		*/
414 		insertNode : function(node) {
415 			var startContainer = this.startContainer,
416 				startOffset = this.startOffset,
417 				endContainer = this.endContainer,
418 				endOffset = this.endOffset,
419 				afterNode,
420 				parentNode,
421 				endNode,
422 				endTextNode,
423 				endTextPos,
424 				eq = startContainer == endContainer,
425 				isFrag = node.nodeName.toLowerCase() === '#document-fragment';
426 			if (endContainer.nodeType == 1 && endOffset > 0) {
427 				endNode = endContainer.childNodes[endOffset - 1];
428 				if (endNode.nodeType == 3) {
429 					eq = startContainer == endNode;
430 					if (eq) endTextPos = endNode.nodeValue.length;
431 				}
432 			}
433 			if (startContainer.nodeType == 1) {
434 				if (startContainer.childNodes.length > 0) {
435 					afterNode = startContainer.childNodes[startOffset];
436 				} else {
437 					parentNode = startContainer;
438 				}
439 			} else {
440 				if (startOffset == 0) {
441 					afterNode = startContainer;
442 				} else if (startOffset < startContainer.length) {
443 					afterNode = startContainer.splitText(startOffset);
444 					if (eq) {
445 						endTextNode = afterNode;
446 						endTextPos = endTextPos ? endTextPos - startOffset : this.endOffset - startOffset;
447 						this.setEnd(endTextNode, endTextPos);
448 					}
449 				} else {
450 					if (startContainer.nextSibling) {
451 						afterNode = startContainer.nextSibling;
452 					} else {
453 						parentNode = startContainer.parentNode;
454 					}
455 				}
456 			}
457 			if (afterNode) afterNode.parentNode.insertBefore(node, afterNode);
458 			if (parentNode) parentNode.appendChild(node);
459 			if (isFrag) {
460 				if (node.firstChild) this.setStartBefore(node.firstChild);
461 				if (this.collapsed) {
462 					if (afterNode) endNode = afterNode.previousSibling;
463 					if (parentNode) endNode = parentNode.lastChild;
464 				}
465 			} else {
466 				this.setStartBefore(node);
467 				if (this.collapsed) endNode = node;
468 			}
469 			if (endNode) this.setEndAfter(endNode);
470 			return this;
471 		},

473 			@name KindEditor.range#surroundContents

507 				range.setEndPoint('EndToStart', _getEndRange(endContainer, endOffset));
508 			} else {
509 				range.setStart(startContainer, startOffset);
510 				range.setEnd(endContainer, endOffset);
511 			}
512 			return range;
513 		},

527 //更新collapsed
528 function _updateCollapsed() {
529 	this.collapsed = (this.startContainer === this.endContainer && this.startOffset === this.endOffset);
530 }
531 //更新commonAncestorContainer
532 function _updateCommonAncestor(doc) {
533 	function getParents(node) {
534 		var parents = [];
535 		while (node) {
536 			parents.push(node);
537 			node = node.parentNode;
538 		}
539 		return parents;
540 	}
541 	var parentsA = getParents(this.startContainer),
542 		parentsB = getParents(this.endContainer),
543 		i = 0, lenA = parentsA.length, lenB = parentsB.length, parentA, parentB;
544 	while (++i) {
545 		parentA = parentsA[lenA - i], parentB = parentsB[lenB - i];
546 		if (!parentA || !parentB || parentA !== parentB) break;
547 	}
548 	this.commonAncestorContainer = parentsA[lenA - i + 1];
549 }
550 //检查开始节点和结束节点的位置，校正错误设置
551 function _compareAndUpdate(doc) {
552 	var rangeA = _range(doc),
553 		rangeB = _range(doc);
554 	rangeA.startContainer = rangeA.endContainer = this.startContainer;
555 	rangeA.startOffset = rangeA.endOffset = this.startOffset;
556 	rangeB.startContainer = rangeB.endContainer = this.endContainer;
557 	rangeB.startOffset = rangeB.endOffset = this.endOffset;
558 	if (rangeA.compareBoundaryPoints(_START_TO_START, rangeB) == 1) {
559 		this.startContainer = this.endContainer;
560 		this.startOffset = this.endOffset;
561 	}
562 }

564 /*
565 	根据参数复制或删除KRange的内容。
566 	cloneContents: copyAndDelete(true, false)
567 	extractContents: copyAndDelete(true, true)
568 	deleteContents: copyAndDelete(false, true)
569 */
570 function _copyAndDelete(doc, isCopy, isDelete) {
571 	var self = this,
572 		startContainer = self.startContainer,
573 		startOffset = self.startOffset,
574 		endContainer = self.endContainer,
575 		endOffset = self.endOffset,
576 		nodeList = [],
577 		selfRange = self;
578 	if (isDelete) {
579 		selfRange = self.cloneRange();
580 		self.collapse(true);
581 		if (startContainer.nodeType == 3 && startOffset == 0) {
582 			self.setStart(startContainer.parentNode, 0);
583 			self.setEnd(startContainer.parentNode, 0);
584 		}
585 	}
586 	function splitTextNode(node, startOffset, endOffset) {
587 		var length = node.nodeValue.length,
588 			centerNode;
589 		if (isCopy) {
590 			var cloneNode = node.cloneNode(true),
591 			centerNode = cloneNode.splitText(startOffset);
592 			centerNode.splitText(endOffset - startOffset);
593 		}
594 		if (isDelete) {
595 			var center = node;
596 			if (startOffset > 0) center = node.splitText(startOffset);
597 			if (endOffset < length) center.splitText(endOffset - startOffset);
598 			nodeList.push(center);
599 		}
600 		return centerNode;
601 	}
602 	function getTextNode(node) {
603 		if (node == startContainer && node == endContainer) {
604 			return splitTextNode(node, startOffset, endOffset);
605 		} else if (node == startContainer) {
606 			return splitTextNode(node, startOffset, node.nodeValue.length);
607 		} else if (node == endContainer) {
608 			return splitTextNode(node, 0, endOffset);
609 		} else {
610 			return splitTextNode(node, 0, node.nodeValue.length);
611 		}
612 	}
613 	function extractNodes(parent, frag) {
614 		var node = parent.firstChild;
615 		while (node) {
616 			var range = _range(doc);
617 			range.selectNode(node);
618 			if (range.compareBoundaryPoints(_END_TO_START, selfRange) >= 0) return false;
619 			var nextNode = node.nextSibling;
620 			if (range.compareBoundaryPoints(_START_TO_END, selfRange) > 0) {
621 				var type = node.nodeType;
622 				if (type == 1) {
623 					if (range.compareBoundaryPoints(_START_TO_START, selfRange) >= 0) {
624 						if (isCopy) {
625 							frag.appendChild(node.cloneNode(true));
626 						}
627 						if (isDelete) {
628 							nodeList.push(node);
629 						}
630 					} else {
631 						var childFlag;
632 						if (isCopy) {
633 							childFlag = node.cloneNode(false);
634 							frag.appendChild(childFlag);
635 						}
636 						if (!extractNodes(node, childFlag)) return false;
637 					}
638 				} else if (type == 3) {
639 					var textNode = getTextNode(node);
640 					if (textNode) frag.appendChild(textNode);
641 				}
642 			}
643 			node = nextNode;
644 		}
645 		return true;
646 	}
647 	var frag = doc.createDocumentFragment(),
648 		ancestor = selfRange.commonAncestorContainer;
649 	if (ancestor.nodeType == 3) {
650 		var textNode = getTextNode(ancestor);
651 		if (textNode) frag.appendChild(textNode);
652 	} else {
653 		extractNodes(ancestor, frag);
654 	}
655 	for (var i = 0, len = nodeList.length; i < len; i++) {
656 		var node = nodeList[i];
657 		node.parentNode.removeChild(node);
658 	}
659 	return isCopy ? frag : self;
660 }
661 //根据原生Range，取得开始节点和结束节点的位置。IE专用
662 function _getStartEnd(rng, isStart) {
663 	var doc = rng.parentElement().ownerDocument;
664 	var range = _range(doc);
665 	var pointRange = rng.duplicate();
666 	pointRange.collapse(isStart);
667 	var parent = pointRange.parentElement();
668 	var children = parent.childNodes;
669 	if (children.length == 0) {
670 		range.selectNode(parent);
671 		return {node: range.startContainer, offset: range.startOffset};
672 	}
673 	var startNode = doc, startPos = 0, isEnd = false;
674 	var testRange = rng.duplicate();
675 	testRange.moveToElementText(parent);
676 	for (var i = 0, len = children.length; i < len; i++) {
677 		var node = children[i];
678 		var cmp = testRange.compareEndPoints('StartToStart', pointRange);
679 		if (cmp > 0) isEnd = true;
680 		if (cmp == 0) {
681 			var range = _range(doc);
682 			if (node.nodeType == 1) range.selectNode(node);
683 			else range.setStartBefore(node);
684 			return {node: range.startContainer, offset: range.startOffset};
685 		}
686 		if (node.nodeType == 1) {
687 			var nodeRange = rng.duplicate();
688 			nodeRange.moveToElementText(node);
689 			testRange.setEndPoint('StartToEnd', nodeRange);
690 			if (isEnd) startPos += nodeRange.text.length;
691 			else startPos = 0;
692 		} else if (node.nodeType == 3) {
693 			testRange.moveStart('character', node.nodeValue.length);
694 			startPos += node.nodeValue.length;
695 		}
696 		if (!isEnd) startNode = node;
697 	}
698 	if (!isEnd && startNode.nodeType == 1) {
699 		range.setStartAfter(parent.lastChild);
700 		return {node: range.startContainer, offset: range.startOffset};
701 	}
702 	testRange = rng.duplicate();
703 	testRange.moveToElementText(parent);
704 	testRange.setEndPoint('StartToEnd', pointRange);
705 	startPos -= testRange.text.length;
706 	return {node: startNode, offset: startPos};
707 }
708 //将原生Range转换成KRange
709 function _toRange(rng) {
710 	if (_IE) {
711 		var doc = rng.parentElement().ownerDocument;
712 		if (rng.item) {
713 			var range = _range(doc);
714 			range.selectNode(rng.item(0));
715 			return range;
716 		}
717 		var start = _getStartEnd(rng, true),
718 			end = _getStartEnd(rng, false),
719 			range = _range(doc);
720 		range.setStart(start.node, start.offset);
721 		range.setEnd(end.node, end.offset);
722 		return range;
723 	} else {
724 		var startContainer = rng.startContainer,
725 			doc = startContainer.ownerDocument || startContainer,
726 			range = _range(doc);
727 		range.setStart(startContainer, rng.startOffset);
728 		range.setEnd(rng.endContainer, rng.endOffset);
729 		return range;
730 	}
731 }
732 //取得父节点里的该节点前的纯文本长度。IE专用
733 function _getBeforeLength(node) {
734 	var doc = node.ownerDocument,
735 		len = 0,
736 		sibling = node.previousSibling;
737 	while (sibling) {
738 		if (sibling.nodeType == 1) {
739 			if (!_node(sibling).isSingle()) {
740 				var range = doc.body.createTextRange();
741 				range.moveToElementText(sibling);
742 				len += range.text.length;
743 			} else {
744 				len += 1;
745 			}
746 		} else if (sibling.nodeType == 3) {
747 			len += sibling.nodeValue.length;
748 		}
749 		sibling = sibling.previousSibling;
750 	}
751 	return len;
752 }
753 //根据Node和offset，取得表示该位置的原生Range。IE专用
754 function _getEndRange(node, offset) {
755 	var doc = node.ownerDocument || node,
756 		range = doc.body.createTextRange();
757 	if (doc == node) {
758 		range.collapse(true);
759 		return range;
760 	}
761 	if (node.nodeType == 1) {
762 		var children = node.childNodes,
763 			isStart,
764 			child,
765 			isTemp = false;
766 		if (offset == 0) {
767 			child = children[0];
768 			isStart = true;
769 		} else {
770 			child = children[offset - 1];
771 			isStart = false;
772 		}
773 		if (!child) {
774 			var temp = doc.createTextNode(' ');
775 			node.appendChild(temp);
776 			child = temp;
777 			isTemp = true;
778 		}
779 		if (child.nodeName.toLowerCase() === 'head') {
780 			if (offset === 1) isStart = true;
781 			if (offset === 2) isStart = false;
782 			range.collapse(isStart);
783 			return range;
784 		}
785 		if (child.nodeType == 1) {
786 			range.moveToElementText(child);
787 			range.collapse(isStart);
788 		} else {
789 			range.moveToElementText(node);
790 			if (isTemp) node.removeChild(temp);
791 			var len = _getBeforeLength(child);
792 			len = isStart ? len : len + child.nodeValue.length;
793 			range.moveStart('character', len);
794 		}
795 	} else if (node.nodeType == 3) {
796 		range.moveToElementText(node.parentNode);
797 		range.moveStart('character', offset + _getBeforeLength(node));
798 	}
799 	return range;
800 }
