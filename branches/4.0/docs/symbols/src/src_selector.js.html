 15 /**
 16 	Only support the following selector syntax.
 17 	Patterns:
 18 	*: any element
 19 	E: an element of type E
 20 	E[foo]: an E element with a "foo" attribute
 21 	E[foo="bar"]: an E element whose "foo" attribute value is exactly equal to "bar"
 22 	E.warning: an E element whose class is "warning" (the document language specifies how class is determined)
 23 	E#myid: an E element with ID equal to "myid"
 24 	E F: an F element descendant of an E element
 25 	E > F: an F element child of an E element
 26 	Reference:
 27 	Selectors Level 3: http://www.w3.org/TR/css3-selectors/
 28 */
 29 (function (K, undefined) {
 30 
 31 var _IE = K.IE,
 32 	_VERSION = K.VERSION,
 33 	_each = K.each,
 34 	_formatCss = K._formatCss,
 35 	_getAttrList = K._getAttrList;
 36 
 37 function _isAncestor(ancestor, node) {
 38 	var parent = node;
 39 	while (parent = parent.parentNode) {
 40 		if (parent === ancestor) return true;
 41 	}
 42 	return false;
 43 }
 44 
 45 function _getAttr(el, key) {
 46 	key = key.toLowerCase();
 47 	var val = null;
 48 	if (_IE && _VERSION < 8) {
 49 		var div = el.ownerDocument.createElement('div');
 50 		div.appendChild(el.cloneNode(false));
 51 		var list = _getAttrList(div.innerHTML.toLowerCase());
 52 		if (key in list) val = list[key];
 53 	} else {
 54 		val = el.getAttribute(key, 2);
 55 	}
 56 	if (key === 'style' && val !== null) {
 57 		val = _formatCss(val);
 58 	}
 59 	return val;
 60 }
 61 
 62 function _query(expr, root) {
 63 	var arr = _queryAll(expr, root);
 64 	return arr.length > 0 ? arr[0] : null;
 65 }
 66 
 67 function _queryAll(expr, root) {
 68 	root = root || document;
 69 	function escape(str) {
 70 		if (typeof str != 'string') return str;
 71 		return str.replace(/([^\w\-])/g, '\\$1');
 72 	}
 73 	function stripslashes(str) {
 74 		return str.replace(/\\/g, '');
 75 	}
 76 	function cmpTag(tagA, tagB) {
 77 		return tagA === '*' || tagA.toLowerCase() === escape(tagB.toLowerCase());
 78 	}
 79 	function byId(id, tag, root) {
 80 		var arr = [];
 81 		var doc = root.ownerDocument || root;
 82 		var el = doc.getElementById(stripslashes(id));
 83 		if (el) {
 84 			if (cmpTag(tag, el.nodeName) && _isAncestor(root, el)) arr.push(el);
 85 		}
 86 		return arr;
 87 	}
 88 	function byClass(className, tag, root) {
 89 		var doc = root.ownerDocument || root;
 90 		var arr = [];
 91 		if (root.getElementsByClassName) {
 92 			var els = root.getElementsByClassName(stripslashes(className));
 93 			for (var i = 0, len = els.length, el; i < len; i++) {
 94 				el = els[i];
 95 				if (cmpTag(tag, el.nodeName)) arr.push(el);
 96 			}
 97 		} else if (doc.querySelectorAll) {
 98 			var els = doc.querySelectorAll((root.nodeName !== '#document' ? root.nodeName + ' ' : '') + tag + '.' + className);
 99 			for (var i = 0, len = els.length, el; i < len; i++) {
100 				el = els[i];
101 				if (_isAncestor(root, el)) arr.push(el);
102 			}
103 		} else {
104 			var els = root.getElementsByTagName(tag);
105 			className = ' ' + className + ' ';
106 			for (var i = 0, len = els.length, el; i < len; i++) {
107 				el = els[i];
108 				if (el.nodeType == 1) {
109 					var cls = el.className;
110 					if (cls && (' ' + cls + ' ').indexOf(className) > -1) arr.push(el);
111 				}
112 			}
113 		}
114 		return arr;
115 	}
116 	function byName(name, tag, root) {
117 		var arr = [];
118 		var els = root.getElementsByName(stripslashes(name));
119 		for (var i = 0, len = els.length, el; i < len; i++) {
120 			el = els[i];
121 			if (cmpTag(tag, el.nodeName)) {
122 				if (el.getAttributeNode('name')) arr.push(el);
123 			}
124 		}
125 		return arr;
126 	}
127 	function byAttr(key, val, tag, root) {
128 		var arr = [];
129 		var els = root.getElementsByTagName(tag);
130 		for (var i = 0, len = els.length, el; i < len; i++) {
131 			el = els[i];
132 			if (el.nodeType == 1) {
133 				if (val === null) {
134 					if (_getAttr(el, key) !== null) arr.push(el);
135 				} else {
136 					if (val === escape(_getAttr(el, key))) arr.push(el);
137 				}
138 			}
139 		}
140 		return arr;
141 	}
142 	function select(expr, root) {
143 		var arr = [], matches;
144 		matches = /^((?:\\.|[^.#\s\[<>])+)/.exec(expr);
145 		var tag = matches ? matches[1] : '*';
146 		if (matches = /#((?:[\w\-]|\\.)+)$/.exec(expr)) {
147 			arr = byId(matches[1], tag, root);
148 		} else if (matches = /\.((?:[\w\-]|\\.)+)$/.exec(expr)) {
149 			arr = byClass(matches[1], tag, root);
150 		} else if (matches = /\[((?:[\w\-]|\\.)+)\]/.exec(expr)) {
151 			arr = byAttr(matches[1].toLowerCase(), null, tag, root);
152 		} else if (matches = /\[((?:[\w\-]|\\.)+)\s*=\s*['"]?((?:\\.|[^'"]+)+)['"]?\]/.exec(expr)) {
153 			var key = matches[1].toLowerCase(), val = matches[2];
154 			if (key === 'id') arr = byId(val, tag, root);
155 			else if (key === 'class') arr = byClass(val, tag, root);
156 			else if (key === 'name') arr = byName(val, tag, root);
157 			else arr = byAttr(key, val, tag, root);
158 		} else {
159 			var els = root.getElementsByTagName(tag);
160 			for (var i = 0, len = els.length, el; i < len; i++) {
161 				el = els[i];
162 				if (el.nodeType == 1) arr.push(el); 
163 			}
164 		}
165 		return arr;
166 	}
167 	var parts = [];
168 	var arr, re = /((?:\\.|[^\s>])+|[\s>])/g;
169 	while (arr = re.exec(expr)) {
170 		if (arr[1] !== ' ') parts.push(arr[1]);
171 	}
172 	var results = [];
173 	if (parts.length == 1) {
174 		return select(parts[0], root);
175 	}
176 	var el, isChild = false;
177 	for (var i = 0, len = parts.length; i < len; i++) {
178 		var part = parts[i];
179 		if (part === '>') {
180 			isChild = true;
181 			continue;
182 		}
183 		if (i > 0) {
184 			var els = [];
185 			_each(results, function(key, val) {
186 				_each(select(part, val), function(k, v) {
187 					if (isChild) {
188 						if (val === v.parentNode) els.push(v);
189 					} else {
190 						els.push(v);
191 					}
192 				});
193 			});
194 			results = els;
195 		} else {
196 			results = select(part, root);
197 		}
198 		if (results.length == 0) return []; 
199 	}
200 	return results;
201 }
202 
203 K._isAncestor = _isAncestor;
204 K._getAttr = _getAttr;
205 
206 K.query = _query;
207 K.queryAll = _queryAll;
208 
209 })(KindEditor);
