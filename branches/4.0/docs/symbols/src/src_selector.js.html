 19 /**
 20 	@name KindEditor.query
 21 	@function
 22 	@param {String} expr 选择器查询语句。
 23 	@param {element} root
 24 	@returns {element}
 25 	@description
 26 	根据expr在root范围内查找DOM元素，并返回第一个元素。没找到则返回null。
 27 	@example
 28 	// Only support the following selector syntax.
 29 	// Patterns:
 30 	// *: any element
 31 	// E: an element of type E
 32 	// E[foo]: an E element with a "foo" attribute
 33 	// E[foo="bar"]: an E element whose "foo" attribute value is exactly equal to "bar"
 34 	// E.warning: an E element whose class is "warning" (the document language specifies how class is determined)
 35 	// E#myid: an E element with ID equal to "myid"
 36 	// E F: an F element descendant of an E element
 37 	// E > F: an F element child of an E element
 38 
 39 	var div = K.query('#id div');
 40 	var span = K.query('span.class', div);
 41 	@see <a href="http://www.w3.org/TR/css3-selectors/" target="_blank">Selectors Level 3</a>
 42 */

 44 /**
 45 	@name KindEditor.queryAll
 46 	@function
 47 	@param {String} expr 选择器查询语句。
 48 	@param {element} root
 49 	@returns {element[]}
 50 	@description
 51 	根据expr在root范围内查找DOM元素，并返回所有元素。没找到则返回空数组。
 52 	@example
 53 	var divArray = K.query('#id div');
 54 	var spanArray = K.query('span.class', div);
 55 */
 56 
 57 (function (K, undefined) {
 58 
 59 var _IE = K.IE,
 60 	_VERSION = K.VERSION,
 61 	_each = K.each,
 62 	_formatCss = K._formatCss,
 63 	_getAttrList = K._getAttrList;
 64 
 65 function _isAncestor(ancestor, node) {
 66 	var parent = node;
 67 	while (parent = parent.parentNode) {
 68 		if (parent === ancestor) return true;
 69 	}
 70 	return false;
 71 }
 72 
 73 function _getAttr(el, key) {
 74 	key = key.toLowerCase();
 75 	var val = null;
 76 	if (_IE && _VERSION < 8) {
 77 		var div = el.ownerDocument.createElement('div');
 78 		div.appendChild(el.cloneNode(false));
 79 		var list = _getAttrList(div.innerHTML.toLowerCase());
 80 		if (key in list) val = list[key];
 81 	} else {
 82 		val = el.getAttribute(key, 2);
 83 	}
 84 	if (key === 'style' && val !== null) {
 85 		val = _formatCss(val);
 86 	}
 87 	return val;
 88 }
 89 
 90 function _query(expr, root) {
 91 	var arr = _queryAll(expr, root);
 92 	return arr.length > 0 ? arr[0] : null;
 93 }
 94 
 95 function _queryAll(expr, root) {
 96 	root = root || document;
 97 	function escape(str) {
 98 		if (typeof str != 'string') return str;
 99 		return str.replace(/([^\w\-])/g, '\\$1');
100 	}
101 	function stripslashes(str) {
102 		return str.replace(/\\/g, '');
103 	}
104 	function cmpTag(tagA, tagB) {
105 		return tagA === '*' || tagA.toLowerCase() === escape(tagB.toLowerCase());
106 	}
107 	function byId(id, tag, root) {
108 		var arr = [];
109 		var doc = root.ownerDocument || root;
110 		var el = doc.getElementById(stripslashes(id));
111 		if (el) {
112 			if (cmpTag(tag, el.nodeName) && _isAncestor(root, el)) arr.push(el);
113 		}
114 		return arr;
115 	}
116 	function byClass(className, tag, root) {
117 		var doc = root.ownerDocument || root;
118 		var arr = [];
119 		if (root.getElementsByClassName) {
120 			var els = root.getElementsByClassName(stripslashes(className));
121 			for (var i = 0, len = els.length, el; i < len; i++) {
122 				el = els[i];
123 				if (cmpTag(tag, el.nodeName)) arr.push(el);
124 			}
125 		} else if (doc.querySelectorAll) {
126 			var els = doc.querySelectorAll((root.nodeName !== '#document' ? root.nodeName + ' ' : '') + tag + '.' + className);
127 			for (var i = 0, len = els.length, el; i < len; i++) {
128 				el = els[i];
129 				if (_isAncestor(root, el)) arr.push(el);
130 			}
131 		} else {
132 			var els = root.getElementsByTagName(tag);
133 			className = ' ' + className + ' ';
134 			for (var i = 0, len = els.length, el; i < len; i++) {
135 				el = els[i];
136 				if (el.nodeType == 1) {
137 					var cls = el.className;
138 					if (cls && (' ' + cls + ' ').indexOf(className) > -1) arr.push(el);
139 				}
140 			}
141 		}
142 		return arr;
143 	}
144 	function byName(name, tag, root) {
145 		var arr = [];
146 		var els = root.getElementsByName(stripslashes(name));
147 		for (var i = 0, len = els.length, el; i < len; i++) {
148 			el = els[i];
149 			if (cmpTag(tag, el.nodeName)) {
150 				if (el.getAttributeNode('name')) arr.push(el);
151 			}
152 		}
153 		return arr;
154 	}
155 	function byAttr(key, val, tag, root) {
156 		var arr = [];
157 		var els = root.getElementsByTagName(tag);
158 		for (var i = 0, len = els.length, el; i < len; i++) {
159 			el = els[i];
160 			if (el.nodeType == 1) {
161 				if (val === null) {
162 					if (_getAttr(el, key) !== null) arr.push(el);
163 				} else {
164 					if (val === escape(_getAttr(el, key))) arr.push(el);
165 				}
166 			}
167 		}
168 		return arr;
169 	}
170 	function select(expr, root) {
171 		var arr = [], matches;
172 		matches = /^((?:\\.|[^.#\s\[<>])+)/.exec(expr);
173 		var tag = matches ? matches[1] : '*';
174 		if (matches = /#((?:[\w\-]|\\.)+)$/.exec(expr)) {
175 			arr = byId(matches[1], tag, root);
176 		} else if (matches = /\.((?:[\w\-]|\\.)+)$/.exec(expr)) {
177 			arr = byClass(matches[1], tag, root);
178 		} else if (matches = /\[((?:[\w\-]|\\.)+)\]/.exec(expr)) {
179 			arr = byAttr(matches[1].toLowerCase(), null, tag, root);
180 		} else if (matches = /\[((?:[\w\-]|\\.)+)\s*=\s*['"]?((?:\\.|[^'"]+)+)['"]?\]/.exec(expr)) {
181 			var key = matches[1].toLowerCase(), val = matches[2];
182 			if (key === 'id') arr = byId(val, tag, root);
183 			else if (key === 'class') arr = byClass(val, tag, root);
184 			else if (key === 'name') arr = byName(val, tag, root);
185 			else arr = byAttr(key, val, tag, root);
186 		} else {
187 			var els = root.getElementsByTagName(tag);
188 			for (var i = 0, len = els.length, el; i < len; i++) {
189 				el = els[i];
190 				if (el.nodeType == 1) arr.push(el); 
191 			}
192 		}
193 		return arr;
194 	}
195 	var parts = [];
196 	var arr, re = /((?:\\.|[^\s>])+|[\s>])/g;
197 	while (arr = re.exec(expr)) {
198 		if (arr[1] !== ' ') parts.push(arr[1]);
199 	}
200 	var results = [];
201 	if (parts.length == 1) {
202 		return select(parts[0], root);
203 	}
204 	var el, isChild = false;
205 	for (var i = 0, len = parts.length; i < len; i++) {
206 		var part = parts[i];
207 		if (part === '>') {
208 			isChild = true;
209 			continue;
210 		}
211 		if (i > 0) {
212 			var els = [];
213 			_each(results, function(key, val) {
214 				_each(select(part, val), function(k, v) {
215 					if (isChild) {
216 						if (val === v.parentNode) els.push(v);
217 					} else {
218 						els.push(v);
219 					}
220 				});
221 			});
222 			results = els;
223 		} else {
224 			results = select(part, root);
225 		}
226 		if (results.length == 0) return []; 
227 	}
228 	return results;
229 }

231 K._isAncestor = _isAncestor;
232 K._getAttr = _getAttr;
233 
234 K.query = _query;
235 K.queryAll = _queryAll;
236 
237 })(KindEditor);
