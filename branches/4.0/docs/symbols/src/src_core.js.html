 76 /**
 77 	@name KindEditor.isArray
 78 	@function
 79 	@param {Mixed} val 要判断的变量
 80 	@returns {Boolean} 变量为数组时返回true，否则返回false。
 81 	@description
 82 	判断一个变量是不是数组。
 83 	@example
 84 	var bool = K.isArray([1, 2, 3]); //返回true
 85 	bool = K.isArray({one : 1}); //返回false
 86 */
 87 function _isArray(val) {
 88 	return Object.prototype.toString.call(val) === '[object Array]';
 89 }

 91 /**
 92 	@name KindEditor.inArray
 93 	@function
 94 	@param {Mixed} val 要查找的变量
 95 	@param {Array} arr 目标数组
 96 	@returns {Int}
 97 	@description
 98 	查找一个变量在一个数组中第一次出现的索引位置。如果没找到，则返回-1。
 99 	@example
100 	var index = K.inArray(2, [1, 2, 3]); //返回1
101 	index = K.inArray(1, [1, 2, 3]); //返回0
102 	index = K.inArray(10, [1, 2, 3]); //返回-1
103 */
104 function _inArray(val, arr) {
105 	for (var i = 0, len = arr.length; i < len; i++) {
106 		if (val === arr[i]) return i;
107 	}
108 	return -1;
109 }

111 /**
112 	@name KindEditor.each
113 	@function
114 	@param {Object|Array} obj 要遍历的对象或数组
115 	@param {function} fn 回调函数，回调函数的第一个参数为key，第二个参数为value。
116 	@returns {undefined}
117 	@description
118 	遍历一个对象或数组。
119 	@example
120 	//遍历数组
121 	K.each([1, 2, 3], function (i, val) {
122 		console.log(i + ':' + val);
123 	});
124 	//遍历对象
125 	K.each({one : 1, two : 2}, function (key, val) {
126 		console.log(key + ':' + val);
127 	});
128 */
129 function _each(obj, fn) {
130 	if (_isArray(obj)) {
131 		for (var i = 0, len = obj.length; i < len; i++) {
132 			if (fn(i, obj[i]) === false) break;
133 		}
134 	} else {
135 		for (var key in obj) {
136 			if (obj.hasOwnProperty(key)) {
137 				if (fn(key, obj[key]) === false) break;
138 			}
139 		}
140 	}
141 }

143 /**
144 	@name KindEditor.trim
145 	@function
146 	@param {String} str
147 	@returns {String}
148 	@description
149 	清除字符串两边的空白。
150 	@example
151 	var str = K.trim(' abc '); //返回"abc"
152 */
153 function _trim(str) {
154 	return str.replace(/^\s+|\s+$/g, '');
155 }

157 /**
158 	@name KindEditor.inString
159 	@function
160 	@param {String} val 要判断的字符串
161 	@param {String} str 用delimiter分隔的目标字符串
162 	@param {String} [delimiter = ","] 分隔符
163 	@returns {Boolean}
164 	@description
165 	判断一个字符串是否包含在目标字符串里。
166 	@example
167 	var bool = K.inString('aaa', 'abc,aaa,bbb,ccc'); //返回true
168 	bool = K.inString('aaa', 'abc aaa bbb ccc', ' '); //返回true
169 */
170 function _inString(val, str, delimiter) {
171 	delimiter = delimiter === undefined ? ',' : delimiter;
172 	return (delimiter + str + delimiter).indexOf(delimiter + val + delimiter) >= 0;
173 }

175 /**
176 	@name KindEditor.toHex
177 	@function
178 	@param {String} color RGB颜色
179 	@returns {String} 16进制颜色
180 	@description
181 	将RGB颜色转换成16进制颜色。
182 	@example
183 	var color = K.toHex('rgb(0, 0, 0)'); //返回"#000000"
184 */
185 function _toHex(color) {
186 	function hex(s) {
187 		s = parseInt(s).toString(16).toUpperCase();
188 		return s.length > 1 ? s : '0' + s;
189 	}
190 	return color.replace(/rgb\s*\(\s*(\d+)\s*,\s*(\d+)\s*,\s*(\d+)\s*\)/ig,
191 		function($0, $1, $2, $3) {
192 			return '#' + hex($1) + hex($2) + hex($3);
193 		}
194 	);
195 }

197 /**
198 	@name KindEditor.toMap
199 	@function
200 	@param {String} str 用delimiter分隔的字符串
201 	@param {String} [delimiter = ","] 分隔符
202 	@returns {Object}
203 	@description
204 	将一个字符串转换成对象。
205 	@example
206 	var map = K.toMap('abc,aaa,bbb'); //返回{abc : true, aaa : true, bbb : true}
207 */
208 function _toMap(str, delimiter) {
209 	var map = {},
210 		delimiter = delimiter === undefined ? ',' : delimiter,
211 		arr = str.split(delimiter);
212 	_each(arr, function(key, val) {
213 		map[val] = true;
214 	});
215 	return map;
216 }
