 53 function _node(expr, root) {
 54 	var node;
 55 	if (typeof expr === 'string') {
 56 		if (/<.+>/.test(expr)) {
 57 			var doc = root ? root.ownerDocument || root : document,
 58 				div = doc.createElement('div');
 59 			div.innerHTML = expr;
 60 			node = div.firstChild;
 61 			div = null;
 62 		} else {
 63 			node = _query(expr, root);
 64 		}
 65 	} else {
 66 		node = expr;
 67 	}
 68 	if (!node) return null;
 69 	var doc = node.ownerDocument || node,
 70 		win = doc.parentWindow || doc.defaultView,
 71 		prevDisplay = '';
 72 	var obj = {
 73 		/**
 74 			@name KindEditor.node#name
 75 			@property
 76 			@public
 77 			@type {String}
 78 			@description
 79 			节点名称。
 80 		*/
 81 		name : node.nodeName.toLowerCase(),
 82 		/**
 83 			@name KindEditor.node#type
 84 			@property
 85 			@public
 86 			@type {String}
 87 			@description
 88 			节点类型。1: Element, 3: textNode
 89 		*/
 90 		type : node.nodeType,
 91 		/**
 92 			@name KindEditor.node#doc
 93 			@property
 94 			@public
 95 			@type {document}
 96 			@description
 97 			包含Node的document对象。
 98 		*/
 99 		doc : doc,
100 		/**
101 			@name KindEditor.node#bind
102 			@function
103 			@public
104 			@param {String} type
105 			@param {String} fn
106 			@returns {KNode}
107 			@description
108 			绑定一个事件。
109 		*/
110 		bind : function(type, fn) {
111 			_bind(node, type, fn);
112 			return this;
113 		},
114 		unbind : function(type, fn) {
115 			_unbind(node, type, fn);
116 			return this;
117 		},
118 		fire : function(type) {
119 			_fire(node, type);
120 			return this;
121 		},
122 		attr : function(key, val) {
123 			if (key === undefined) {
124 				return _getAttrList(this.outer());
125 			} else if (val === undefined) {
126 				val = _getAttr(node, key);
127 				return val === null ? '' : val;
128 			} else {
129 				if (_IE && _VERSION < 8 && key.toLowerCase() == 'class') key = 'className';
130 				node.setAttribute(key, '' + val);
131 				return this;
132 			}
133 		},
134 		removeAttr : function(key) {
135 			if (_IE && _VERSION < 8 && key.toLowerCase() == 'class') key = 'className';
136 			this.attr(key, '');
137 			node.removeAttribute(key);
138 			return this;
139 		},
140 		get : function() {
141 			return node;
142 		},
143 		hasClass : function(cls) {
144 			return _inString(cls, node.className, ' ');
145 			
146 		},
147 		addClass : function(cls) {
148 			if (!this.hasClass(cls)) node.className = _trim(node.className + ' ' + cls);
149 			return this;
150 		},
151 		removeClass : function(cls) {
152 			if(this.hasClass(cls))
153 				node.className = _trim(node.className.replace(new RegExp('\s*' + cls + '\s*'), ''));
154 			return this;
155 		},
156 		html : function(val) {
157 			if (val === undefined) {
158 				return _formatHtml(node.innerHTML);
159 			} else {
160 				node.innerHTML = _formatHtml(val);
161 				return this;
162 			}
163 		},
164 		val : function(val) {
165 			return this.attr('value' , val);
166 		},
167 		css : function(key, val) {
168 			var self = this;
169 			if (key === undefined) {
170 				return _getCssList(this.attr('style'));
171 			}
172 			if (typeof key === 'object') {
173 				_each(key, function(k, v) {
174 					self.css(k, v);
175 				});
176 				return this;
177 			}
178 			if (val === undefined) {
179 				return node.style[key] || this.computedCss(key) || '';
180 			}
181 			node.style[_toCamel(key)] = val;
182 			return this;
183 		},
184 		computedCss : function(key) {
185 			var camelKey = _toCamel(key),
186 				val = '';
187 			if (win.getComputedStyle) {
188 				var style = win.getComputedStyle(node, null);
189 				val = style[camelKey] || style.getPropertyValue(key) || node.style[camelKey];
190 			} else if (node.currentStyle) {
191 				val = node.currentStyle[camelKey] || node.style[camelKey];
192 			}
193 			return val;
194 		},
195 		clone : function(bool) {
196 			return _node(node.cloneNode(bool));
197 		},
198 		append : function(val) {
199 			node.appendChild(_get(val));
200 			return this;
201 		},
202 		before : function(val) {
203 			node.parentNode.insertBefore(_get(val), node);
204 			return this;
205 		},
206 		after : function(val) {
207 			if (node.nextSibling) {
208 				node.parentNode.insertBefore(_get(val), node.nextSibling);
209 			} else {
210 				this.append(val);
211 			}
212 			return this;
213 		},
214 		replaceWith : function(val) {
215 			node.parentNode.replaceChild(_get(val), node);
216 			this.unbind();
217 			node = _get(val);
218 			return this;
219 		},
220 		remove : function() {
221 			if(_IE){
222 				var div = document.createElement('div');
223 				div.appendChild(node.get());
224 				div.innerHTML = null;
225 				div = null;
226 			}else{
227 				ndoe.parentNode.removeChild(node);
228 			}
229 			this.unbind();
230 			node = null;
231 			return this;
232 		},
233 		show : function() {
234 			if (this.computedCss('display') === 'none') 
235 				this.css('display', prevDisplay);
236 			return this;
237 		},
238 		hide : function() {
239 			if (this.computedCss('display') !== 'none') {
240 				prevDisplay = this.css('display');
241 				this.css('display', 'none');
242 			}
243 			return this;
244 		},
245 		outer : function() {
246 			var div = doc.createElement('div'),html;
247 			div.appendChild(node);
248 			html = div.innerHTML;
249 			div = null;
250 			return html;
251 		},
252 		isSingle : function() {
253 			return !!_SINGLE_TAG_MAP[this.name];
254 		},
255 		isInline : function() {
256 			return !!_INLINE_TAG_MAP[this.name];
257 		},
258 		isBlock : function() {
259 			return !!_BLOCK_TAG_MAP[this.name];
260 		},
261 		isAncestor : function(ancestor) {
262 			return _isAncestor(_get(ancestor), node);
263 		},
264 		parent : function() {
265 			return _node(node.parentNode);
266 		},
267 		prev : function() {
268 			return _node(node.previousSibling);
269 		},
270 		next : function() {
271 			return _node(node.nextSibling);
272 		},
273 		each : function(fn, order) {
274 			order = (order === undefined) ? true : order;
275 			function walk(knode) {
276 				var n = order ? knode.first : knode.last;
277 				if (!n) return;
278 				while (n) {
279 					var next = order ? n.next() : n.prev();
280 					if (fn(n)) return true;
281 					if (walk(n)) return;
282 					n = next;
283 				}
284 			}
285 			walk(this);
286 		},
287 		toString : function() {
288 			return this.type == 3 ? node.nodeValue : this.outer();
289 		}
290 	};
291 	_updateProp.call(obj, node);
292 	return obj;
293 }

299 function _toCamel(str) {
300 	var arr = str.split('-');
301 	str = '';
302 	_each(arr, function(key, val) {
303 		str += (key > 0) ? val.charAt(0).toUpperCase() + val.substr(1) : val;
304 	});
305 	return str;
306 }

308 function _updateProp(node) {
309 	//node.first, node.last, node.children
310 	var list = [], child = node.firstChild;
311 	while (child) {
312 		if (child.nodeType != 3 || _trim(child.nodeValue) !== '') {
313 			list.push(_node(child));
314 		}
315 		child = child.nextSibling;
316 	}
317 	if (list.length > 0) {
318 		this.first = list[0];
319 		this.last = list[list.length - 1];
320 	} else {
321 		this.first = this.last = null;
322 	}
323 	this.children = list;
324 	//node.index
325 	var i = -1, sibling = node;
326 	while (sibling) {
327 		i++;
328 		sibling = sibling.previousSibling;
329 	}
330 	this.index = i;
331 }
