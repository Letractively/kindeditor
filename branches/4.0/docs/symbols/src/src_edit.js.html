 14 /**
 15 #using "core.js"
 16 #using "node.js"
 17 #using "cmd.js"
 18 */
 19 (function (K, undefined) {
 20 
 21 var _each = K.each,
 22 	_isArray = K.isArray,
 23 	_node = K.node,
 24 	_cmd = K.cmd;
 25 
 26 function _getIframeDoc(iframe) {
 27 	return iframe.contentDocument || iframe.contentWindow.document;
 28 }
 29 
 30 function _getInitHtml(bodyClass, css) {
 31 	var arr = ['<!doctype html><html><head><meta charset="utf-8" /><title>KindEditor</title>'];
 32 	if (css) {
 33 		if (typeof css == 'string' && !/\{[\s\S]*\}/g.test(css)) {
 34 			css = [css];
 35 		}
 36 		if (_isArray(css)) {
 37 			_each(css, function(i, path) {
 38 				if (path !== '') arr.push('<link href="' + path + '" rel="stylesheet" />');
 39 			});
 40 		} else {
 41 			arr.push('<style>' + css + '</style>');
 42 		}
 43 	}
 44 	arr.push('</head><body ' + (bodyClass ? 'class="' + bodyClass + '"' : '') + '></body></html>');
 45 	return arr.join('');
 46 }
 47 
 48 function _iframeVal(val) {
 49 	var self = this,
 50 		body = self.doc.body;
 51 	if (val === undefined) {
 52 		return _node(body).html();
 53 	} else {
 54 		_node(body).html(val);
 55 		return self;
 56 	}
 57 }
 58 
 59 function _textareaVal(val) {
 60 	var self = this,
 61 		textarea = self.textarea;
 62 	if (val === undefined) {
 63 		return textarea.val();
 64 	} else {
 65 		textarea.val(val);
 66 		return self;
 67 	}
 68 }
 69 
 70 /**
 71 	@name KindEditor.edit
 72 	@class 可视化控件类
 73 	@param {String|Node} expr DOM元素、选择器字符串
 74 	@param {Object} options 可选配置
 75 	@description
 76 	可视化控件类。
 77 	@example
 78 	K.edit('textarea').create(); //将第一个匹配的textarea转换成可视化控件
 79 	K.edit(document.getElementById('textarea_id')).create();
 80 */
 81 function _edit(expr, options) {
 82 	var srcElement = _node(expr),
 83 		designMode = options.designMode === undefined ? true : options.designMode,
 84 		bodyClass = options.bodyClass,
 85 		css = options.css;
 86 	function srcVal(val) {
 87 		return srcElement.hasVal() ? srcElement.val(val) : srcElement.html(val);
 88 	}
 89 	var obj = {
 90 		/**
 91 			@name KindEditor.edit#cmd
 92 			@property
 93 			@public
 94 			@type {KCmd}
 95 			@description
 96 			可视化控件的cmd对象，可以执行cmd下的命令。
 97 		*/
 98 		/**
 99 			@name KindEditor.edit#doc
100 			@property
101 			@public
102 			@type {iframeDocument}
103 			@description
104 			可视化控件iframe的document对象。
105 		*/
106 		/**
107 			@name KindEditor.edit#iframe
108 			@property
109 			@public
110 			@type {iframeElement}
111 			@description
112 			可视化控件的iframe对象。
113 		*/
114 		/**
115 			@name KindEditor.edit#textarea
116 			@property
117 			@public
118 			@type {textareaElement}
119 			@description
120 			可视化控件的textarea对象。
121 		*/
122 		width : options.width || 0,
123 		height : options.height || 0,
124 		val : function(val) {
125 			if (designMode) {
126 				return _iframeVal.call(this, val);
127 			} else {
128 				return _textareaVal.call(this, val);
129 			}
130 		},
131 		create : function() {
132 			var self = this;
133 			if (self.iframe) return self;
134 			//create elements
135 			var iframe = _node('<iframe class="ke-iframe" frameborder="0"></iframe>');
136 			iframe.css({
137 				display : 'block',
138 				width : self.width,
139 				height : self.height
140 			});
141 			var textarea = _node('<textarea class="ke-textarea"></textarea>');
142 			textarea.css({
143 				display : 'block',
144 				width : self.width,
145 				height : self.height
146 			});
147 			if (designMode) textarea.hide()
148 			else iframe.hide();
149 			srcElement.before(iframe);
150 			srcElement.before(textarea);
151 			srcElement.hide();
152 			var doc = _getIframeDoc(iframe.get());
153 			doc.designMode = 'on';
154 			doc.open();
155 			doc.write(_getInitHtml(bodyClass, css));
156 			doc.close();
157 			self.iframe = iframe;
158 			self.textarea = textarea;
159 			self.doc = doc;
160 			if (designMode) _iframeVal.call(self, srcVal());
161 			else _textareaVal.call(self, srcVal());
162 			self.cmd = _cmd(doc);
163 			//add events
164 			function selectionHandler(e) {
165 				var cmd = _cmd(doc);
166 				if (cmd) self.cmd = cmd;
167 			}
168 			self.oninput(selectionHandler);
169 			_node(doc).bind('mouseup', selectionHandler);
170 			_node(document).bind('mousedown', selectionHandler);
171 			return self;
172 		},
173 		remove : function() {
174 			var self = this,
175 				iframe = self.iframe,
176 				textarea = self.textarea,
177 				doc = self.doc;
178 			if (!iframe) return self;
179 			//remove events
180 			_node(doc).unbind();
181 			_node(doc.body).unbind();
182 			_node(document).unbind();
183 			//remove elements
184 			srcElement.show();
185 			srcVal(self.val());
186 			doc.src = 'javascript:false';
187 			iframe.remove();
188 			textarea.remove();
189 			self.iframe = self.textarea = null;
190 			return self;
191 		},
192 		toggle : function(bool) {
193 			var self = this,
194 				iframe = self.iframe,
195 				textarea = self.textarea;
196 			if (!iframe) return self;
197 			if (bool === undefined ? !designMode : bool) {
198 				if (!designMode) {
199 					textarea.hide();
200 					_iframeVal.call(self, _textareaVal.call(self));
201 					iframe.show();
202 					designMode = true;
203 				}
204 			} else {
205 				if (designMode) {
206 					iframe.hide();
207 					_textareaVal.call(self, _iframeVal.call(self));
208 					textarea.show();
209 					designMode = false;
210 				}
211 			}
212 			return self;
213 		},
214 		toDesign : function() {
215 			return this.toggle(true);
216 		},
217 		toSource : function() {
218 			return this.toggle(false);
219 		},
220 		show : function() {
221 			var self = this;
222 			self.iframe && self.iframe.show();                             
223 			return self;
224 		},
225 		hide : function() {
226 			var self = this;
227 			self.iframe && self.iframe.hide();
228 			return self;
229 		},
230 		focus : function() {
231 			var self = this;
232 			self.iframe && self.iframe.contentWindow.focus();
233 			return self;
234 		},
235 		oninput : function(fn) {
236 			var self = this,
237 				doc = self.doc,
238 				body = doc.body;
239 			_node(doc).bind('keyup', function(e) {
240 				if (!e.ctrlKey && !e.altKey && (e.keyCode < 16 || e.keyCode > 18) && e.keyCode != 116) {
241 					fn(e);
242 					e.stop();
243 				}
244 			});
245 			function timeoutHandler(e) {
246 				setTimeout(function() {
247 					fn(e);
248 				}, 1);
249 			}
250 			_node(body).bind('paste', timeoutHandler);
251 			_node(body).bind('cut', timeoutHandler);
252 			return self;
253 		}
254 	};
255 	return obj;
256 }
257 
258 K.edit = _edit;
259 
260 })(KindEditor);
