 10 /**
 11 #using "core.js"
 12 #using "html.js"

 14 #using "node.js"

 17 (function (K, undefined) {
 18 
 19 var _each = K.each,
 20 	_node = K.node,
 21 	_range = K.range,
 22 	_IE = K.IE,
 23 	_INLINE_TAGS = K.INLINE_TAGS;
 24 
 25 //get window by document
 26 function _getWin(doc) {
 27 	return doc.parentWindow || doc.defaultView;
 28 }
 29 //get current selection of a document
 30 function _getSel(doc) {
 31 	var win = _getWin(doc);
 32 	return win.getSelection ? win.getSelection() : doc.selection;
 33 }
 34 /**
 35 	Examples:
 36 	K.cmd(document).bold();
 37 	K.cmd(document).wrap('<span style="color:red;"></span>');
 38 	K.cmd(document).remove({
 39 		span : '*',
 40 		div : 'class,border'
 41 	});
 42 */
 43 function _cmd(mixed) {
 44 	var sel, doc, rng;
 45 	if (mixed.nodeName) {
 46 		//get selection and original range when mixed is a document or a node
 47 		doc = mixed.nodeName.toLowerCase() === '#document' ? mixed : mixed.ownerDocument;
 48 		sel = _getSel(doc);
 49 		try {
 50 			if (sel.rangeCount > 0) rng = sel.getRangeAt(0);
 51 			else rng = sel.createRange();
 52 		} catch(e) {}
 53 		mixed = rng || doc;
 54 		if (_IE) {
 55 			if (rng.parentElement().ownerDocument !== doc) return null;
 56 		}
 57 	} else {
 58 		//get selection and original range when mixed is K.range
 59 		var startContainer = mixed.startContainer;
 60 		doc = startContainer.ownerDocument || startContainer;
 61 		sel = _getSel(doc);
 62 		rng = mixed.get();
 63 	}
 64 	var win = _getWin(doc);
 65 	var range = _range(mixed);
 66 	//select a K.range (add K.range to the selection)
 67 	function _select(range) {
 68 		var rng = range.get();
 69 		if (_IE) rng.select();
 70 		else sel.addRange(rng);
 71 		win.focus();
 72 	}
 73 	//new K.range object
 74 	return {
 75 		wrap : function(val) {
 76 			var wrapper = _node(val, doc),
 77 			name = wrapper.name,
 78 			frag = range.extractContents();
 79 			_node(frag).each(function(node) {
 80 				if (node.type == 3 && node.parent().name !== name) {
 81 					var clone = wrapper.clone(false);
 82 					clone.append(node.clone(true));
 83 					node.replaceWith(clone);
 84 				} else if (node.name === name) {
 85 					_each(wrapper.attr(), function(key, val) {
 86 						if (key !== 'style') node.attr(key, val);
 87 					});
 88 					_each(wrapper.css(), function(key, val) {
 89 						node.css(key, val);
 90 					});
 91 				}
 92 			});
 93 			range.insertNode(frag);
 94 			_select(range);
 95 		},
 96 		remove : function(options) {
 97 			_select(range);
 98 		},
 99 		bold : function() {
100 			this.wrap('<strong></strong>');
101 		},
102 		italic : function() {
103 			this.wrap('<em></em>');
104 		},
105 		foreColor : function(val) {
106 			this.wrap('<span style="color:' + val + ';"></span>');
107 		},
108 		hiliteColor : function() {
109 			this.wrap('<span style="background-color:' + val + ';"></span>');
110 		},
111 		removeFormat : function() {
112 			var options = {
113 				'*' : 'class,style'
114 			},
115 			tags = _INLINE_TAGS.split(',');
116 			_each(tags, function(key, val) {
117 				options[val] = '*';
118 			});
119 			this.remove(options);
120 		}
121 	};
122 }
123 
124 function _execCommand(doc, cmdName, ui, val) {
125 	var cmd = _cmd(doc);
126 	cmd[cmdName].call(cmd, val);
127 }
128 
129 function _queryCommandValue(doc, cmdName) {
130 
131 }
132 
133 K.cmd = _cmd;
134 K.execCommand = _execCommand;
135 K.queryCommandValue = _queryCommandValue;
136 
137 })(KindEditor);
